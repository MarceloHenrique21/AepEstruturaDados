#include "gerenciamento.h"

int menu(){
	
	int esc;
	
	LPedido P;
	
	do{

		printf("\n ------------MENU------------");
		printf("\n 1 - Inserir o produto");
		printf("\n 2 - Apresentar os produtos");
		printf("\n 3 - Excluir algum produto");
		printf("\n 0 - Sair do menu");
		printf("\n ----------------------------");
		printf("\n Escolha uma opcao: ");

		scanf("%d", &esc);
		switch(esc){
			case 1: inserirProduto(&P);	break;
			case 2: apresentarProdutos(&P); break;
			case 3: excluirProdutos(&P); 	break;
		}		
	}
		while(esc !=0);
		
}
